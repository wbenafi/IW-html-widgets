Es posible crear un reloj funcional si se utiliza javascript para asignar las variables cada segundo 
function setTime() {
  var time = new Date(),
  second = time.getSeconds();
  minute = time.getMinutes();
  hour = time.getHours();
  console.log(second);
  console.log(minute);
  console.log(hour);
  document.querySelector('.clock').style = `--sec:${second}; --min:${minute}; --hr:${hour}`;
}
var interval = window.setInterval(setTime,1000);
